## perubahan di ios build
- pod nya diubah ke 
```
source 'https://cdn.cocoapods.org/'
source 'https://github.com/CocoaPods/Specs'

platform :ios, '15.0'

target 'UnityFramework' do
  pod 'Firebase/Analytics', '10.17.0'
  pod 'Firebase/Core', '10.17.0'
  pod 'Firebase/RemoteConfig', '10.17.0'
  pod 'Google-Mobile-Ads-SDK', '11.5.0'
  pod 'GoogleUserMessagingPlatform', '2.6.0'
  pod 'OneSignalXCFramework', '5.2.4'
end
target 'Unity-iPhone' do
  pod 'OneSignalXCFramework', '5.2.4'
end
use_frameworks! :linkage => :static
target 'OneSignalNotificationServiceExtension' do
  pod 'OneSignalXCFramework', '5.2.4'
end

```
- swift dummynya diubah
```
// Generated by External Dependency Manager for Unity
import Foundation
import StoreKit

@available(iOS 15.0, *)
@objc class DummyStoreKitLinker: NSObject {
    func forceLinkStoreKit2Symbols() {
        // Pancing enum dan tipe Swift StoreKit 2
        let _: Product.SubscriptionOffer.PaymentMode = .freeTrial
        let _: Product.SubscriptionOffer.PaymentMode = .payAsYouGo
        let _: Product.SubscriptionOffer.PaymentMode = .payUpFront
    }
}

```

- Unity framework - build setting
```
Always Embed Swift Standard Libraries → YES

Enable Modules (C and Objective-C) → YES

Defines Module → YES
```


- Pod install dengan clear
```
rm -rf Pods Podfile.lock *.xcworkspace
pod install --repo-update
```
